/******++++**************************** EJERCICIO perdio ***********+++***********************/

bool perdio(tablero& t, jugadas& j) {
    bool res=false;
    for(int i=0;i<j.size();i++){
        if(t[get<0>(get<0>(j[i]))][get<1>(get<0>(j[i]))]==true){
            res= true;
        }
    }
    return res;
}
